#Set up environment variables
RG=<your-resource-group-name>
Vnet=containerslabl100net
Subnet=containerslabl100sub


#Create Resource Group
az group create --name $RG --location westus

#Create Azure Virtual Network
az network vnet create \
    --resource-group $RG \
    --name $Vnet \
    --address-prefixes 10.0.0.0/8 \
    --subnet-name $Subnet \
    --subnet-prefix 10.240.0.0/16

#Create master VM
az vm create \
    --resource-group $RG \
    --name k8s-master \
    --image UbuntuLTS \
    --admin-username azureuser \
    --generate-ssh-keys \
    --vnet-name $Vnet \
    --subnet $Subnet \
    --size Standard_D2s_v3 \
    --os-disk-size-gb 50

#Create worker VM
az vm create \
    --resource-group $RG \
    --name k8s-worker \
    --image UbuntuLTS \
    --admin-username azureuser \
    --generate-ssh-keys \
    --vnet-name $Vnet \
    --subnet $Subnet \
    --size Standard_D2s_v3 \
    --os-disk-size-gb 50

# INSTALL cks-master
ssh azureuser@$(az vm list-ip-addresses -n k8s-master -g $RG --query [0].virtualMachine.network.publicIpAddresses[0].ipAddress -o tsv)
sudo -i
bash <(curl -s https://raw.githubusercontent.com/killer-sh/cks-course-environment/master/cluster-setup/latest/install_master.sh)


# INSTALL cks-worker
ssh azureuser@$(az vm list-ip-addresses -n k8s-worker -g $RG --query [0].virtualMachine.network.publicIpAddresses[0].ipAddress -o tsv)
sudo -i
bash <(curl -s https://raw.githubusercontent.com/killer-sh/cks-course-environment/master/cluster-setup/latest/install_worker.sh)
