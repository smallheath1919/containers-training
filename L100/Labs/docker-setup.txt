#Set up environment variables
RG=<your-resource-group-name>

#Create docker VM
az vm create \
    --resource-group $RG \
    --name dockerlab \
    --image UbuntuLTS \
    --admin-username azureuser \
    --generate-ssh-keys \
    --size Standard_D2s_v3 \
    --os-disk-size-gb 50

#Login to the docker VM
ssh azureuser@$(az vm list-ip-addresses -n dockerlab -g $RG --query [0].virtualMachine.network.publicIpAddresses[0].ipAddress -o tsv)

#Set up docker
sudo -i
apt-get update
apt-get install \
    apt-transport-https \
    ca-certificates \
    curl \
    gnupg \
    lsb-release
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
echo \
  "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
apt-get update
apt-get install docker-ce docker-ce-cli containerd.io git
